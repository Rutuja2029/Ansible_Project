Modules:

# to create directory
ansible.builtin.file:
  path
  state: directory

# to create file
ansible.builtin.file:
  src:
  state: touch

# to create multiple file using loop
ansible.builtin.file:
  src: /path/{{ item }}
  state: touch
with_items:
  - item1
  - item2
  - item3
when: var == "Name"

  dest:
